h = pow(1010 * p + 1011, q, n)
h=(1010 * p + 1011)^q%n
h=(1010 * p + 1011)^q+kn
h=(1011)^q+kn+kp
h%p=(1011)^q%p


n//q=p
n//q-1=p-1
n-q=(p-1)*q
q=n-(p-1)*q
因为q=n-(p-1)*q

h%p=(1011)^(n-(p-1)*q)%p

h%p=[(1011)^n*1011^(-q)*(p-1)]%p
h%p=[(1011)^n%p*1011^(-q)*(p-1)%p]%p
h%p=[(1011)^n%p*(1011^-q)^(p-1)%p]%p
#根据费马小定理
h%p=[(1011)^n%p]%p

h%p=(1011)^n%p
kp=1011^n-h
所以利用公约数 求出p
p=gcd(1011^n-h,n)
